150  def loadBase64Img(uri):
151     encoded_data = uri.split(',')[1]
152     nparr = np.fromstring(base64.b64decode(encoded_data), np.uint8)
153     img = cv2.imdecode(nparr, cv2.IMREAD_COLOR)
154     return img

189  def detect_face(img, detector_backend = 'opencv', grayscale = False, enforce_detection = True):
190  	
191  	home = str(Path.home())
192  	
193  	#if functions.preproces_face is called directly, then face_detector global variable might not been initialized.
194  	if not "face_detector" in globals():
195  		initialize_detector(detector_backend = detector_backend)
196  		
197  	if detector_backend == 'opencv':
198  		
199  		faces = []
200  		
201  		try: 
202  			faces = face_detector.detectMultiScale(img, 1.3, 5)
203  		except:
204  			pass
205  		
206  		if len(faces) > 0:
207  			x,y,w,h = faces[0] #focus on the 1st face found in the image
208  			detected_face = img[int(y):int(y+h), int(x):int(x+w)]
209  			return detected_face
210  		
211  		else: #if no face detected
212  	
213  			if enforce_detection != True:			
214  				return img
215  	
216  			else:
217  				raise ValueError("Face could not be detected. Please confirm that the picture is a face photo or consider to set enforce_detection param to False.")
218  
219  	elif detector_backend == 'ssd':
220  		
221  		ssd_labels = ["img_id", "is_face", "confidence", "left", "top", "right", "bottom"]
222  		
223  		target_size = (300, 300)
224  		
225  		base_img = img.copy() #we will restore base_img to img later
226  		
227  		original_size = img.shape
228  		
229  		img = cv2.resize(img, target_size)
230  		
231  		aspect_ratio_x = (original_size[1] / target_size[1])

234  		imageBlob = cv2.dnn.blobFromImage(image = img)
235  		
236  		face_detector.setInput(imageBlob)
237  		detections = face_detector.forward()

240  		
241  		detections_df = detections_df[detections_df['is_face'] == 1] #0: background, 1: face
242  		detections_df = detections_df[detections_df['confidence'] >= 0.90]
243  		
244  		detections_df['left'] = (detections_df['left'] * 300).astype(int)
245  		detections_df['bottom'] = (detections_df['bottom'] * 300).astype(int)
246  		detections_df['right'] = (detections_df['right'] * 300).astype(int)
247  		detections_df['top'] = (detections_df['top'] * 300).astype(int)
248  		
249  		if detections_df.shape[0] > 0:
250  			
251  			#TODO: sort detections_df
252  			
253  			#get the first face in the image    
254  			instance = detections_df.iloc[0]
255  			
256  			left = instance["left"]
257  			right = instance["right"]
258  			bottom = instance["bottom"]
259  			top = instance["top"]
260  			
261  			detected_face = base_img[int(top*aspect_ratio_y):int(bottom*aspect_ratio_y), int(left*aspect_ratio_x):int(right*aspect_ratio_x)]
262  			
263  			return detected_face
264  			
265  		else: #if no face detected
266  	
267  			if enforce_detection != True:
268  				img = base_img.copy()
269  				return img
270  	
271  			else:
272  				raise ValueError("Face could not be detected. Please confirm that the picture is a face photo or consider to set enforce_detection param to False.")
273  	
274  	elif detector_backend == 'dlib':
275  
276  		detections = face_detector(img, 1)
277  		
278  		if len(detections) > 0:
279  			
280  			for idx, d in enumerate(detections):
281  				left = d.left(); right = d.right()
282  				top = d.top(); bottom = d.bottom()
283  				
284  				detected_face = img[top:bottom, left:right]
285  				
286  				return detected_face
287  			
288  		else: #if no face detected
289  	
290  			if enforce_detection != True:			
291  				return img
292  	
293  			else:
294  				raise ValueError("Face could not be detected. Please confirm that the picture is a face photo or consider to set enforce_detection param to False.") 
295  		
296  	elif detector_backend == 'mtcnn':
297  		
298  		img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB) #mtcnn expects RGB but OpenCV read BGR
299  		detections = face_detector.detect_faces(img_rgb)
300  		
301  		if len(detections) > 0:
302  			detection = detections[0]
303  			x, y, w, h = detection["box"]
304  			detected_face = img[int(y):int(y+h), int(x):int(x+w)]
305  			return detected_face
306  		
307  		else: #if no face detected
308  			if not enforce_detection:			
309  				return img
310  	
311  			else:
312  				raise ValueError("Face could not be detected. Please confirm that the picture is a face photo or consider to set enforce_detection param to False.")
313  	
314  	else:
315  		detectors = ['opencv', 'ssd', 'dlib', 'mtcnn']
316  		raise ValueError("Valid backends are ", detectors," but you passed ", detector_backend)

318  def alignment_procedure(img, left_eye, right_eye):
319  		
320  	#this function aligns given face in img based on left and right eye coordinates
321  	
322  	left_eye_x, left_eye_y = left_eye
323  	right_eye_x, right_eye_y = right_eye
324  	
325  	#-----------------------
326  	#find rotation direction
327  		
328  	if left_eye_y > right_eye_y:
329  		point_3rd = (right_eye_x, left_eye_y)
330  		direction = -1 #rotate same direction to clock
331  	else:
332  		point_3rd = (left_eye_x, right_eye_y)
333  		direction = 1 #rotate inverse direction of clock
334  	
335  	#-----------------------
336  	#find length of triangle edges
337  	
338  	a = distance.findEuclideanDistance(np.array(left_eye), np.array(point_3rd))
339  	b = distance.findEuclideanDistance(np.array(right_eye), np.array(point_3rd))
340  	c = distance.findEuclideanDistance(np.array(right_eye), np.array(left_eye))
341  	
342  	#-----------------------
343  	
344  	#apply cosine rule
345  			
346  	if b != 0 and c != 0: #this multiplication causes division by zero in cos_a calculation
347  		
348  		cos_a = (b*b + c*c - a*a)/(2*b*c)
349  		angle = np.arccos(cos_a) #angle in radian
350  		angle = (angle * 180) / math.pi #radian to degree
351  		
352  		#-----------------------
353  		#rotate base image
354  		
355  		if direction == -1:
356  			angle = 90 - angle
357  		
358  		img = Image.fromarray(img)
359  		img = np.array(img.rotate(direction * angle))
360  	
361  	#-----------------------
362  	
363  	return img #return img anyway

365  def align_face(img, detector_backend = 'opencv'):
366  	
367  	home = str(Path.home())
368  	
369  	if (detector_backend == 'opencv') or (detector_backend == 'ssd'):
370  		
371  		detected_face_gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY) #eye detector expects gray scale image
372  		
373  		eyes = eye_detector.detectMultiScale(detected_face_gray)
374  		
375  		if len(eyes) >= 2:
376  			
377  			#find the largest 2 eye
378  			
379  			base_eyes = eyes[:, 2]
380  			
381  			items = []
382  			for i in range(0, len(base_eyes)):
383  				item = (base_eyes[i], i)
384  				items.append(item)
385  			
386  			df = pd.DataFrame(items, columns = ["length", "idx"]).sort_values(by=['length'], ascending=False)
387  			
388  			eyes = eyes[df.idx.values[0:2]] #eyes variable stores the largest 2 eye
389  			
390  			#-----------------------
391  			#decide left and right eye
392  			
393  			eye_1 = eyes[0]; eye_2 = eyes[1]
394  			
395  			if eye_1[0] < eye_2[0]:
396  				left_eye = eye_1; right_eye = eye_2
397  			else:
398  				left_eye = eye_2; right_eye = eye_1
399  			
400  			#-----------------------
401  			#find center of eyes
402  			
403  			left_eye = (int(left_eye[0] + (left_eye[2] / 2)), int(left_eye[1] + (left_eye[3] / 2)))
404  			right_eye = (int(right_eye[0] + (right_eye[2]/2)), int(right_eye[1] + (right_eye[3]/2)))
405  			
406  			img = alignment_procedure(img, left_eye, right_eye)
407  			
408  		return img #return img anyway
409  	
410  	elif detector_backend == 'dlib':
411  		
412  		import dlib #this is not a must dependency in deepface
413  		
414  		detections = face_detector(img, 1)
415  		
416  		if len(detections) > 0:
417  			detected_face = detections[0]
418  			img_shape = sp(img, detected_face)
419  			img = dlib.get_face_chip(img, img_shape, size = img.shape[0])
420  			
421  		return img #return img anyway
422  	
423  	elif detector_backend == 'mtcnn':
424  		
425  		img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB) #mtcnn expects RGB but OpenCV read BGR
426  		detections = face_detector.detect_faces(img_rgb)
427  		
428  		if len(detections) > 0:
429  			detection = detections[0]
430  			
431  			keypoints = detection["keypoints"]
432  			left_eye = keypoints["left_eye"]
433  			right_eye = keypoints["right_eye"]
434  			
435  			img = alignment_procedure(img, left_eye, right_eye)
436  				
437  		return img #return img anyway

439  def preprocess_face(img, target_size=(224, 224), grayscale = False, enforce_detection = True, detector_backend = 'opencv'):
440  	
441  	#img_path = copy.copy(img)
442  	
443  	#img might be path, base64 or numpy array. Convert it to numpy whatever it is.
444  	img = load_image(img)
445  	base_img = img.copy()
446  	
447  	img = detect_face(img = img, detector_backend = detector_backend, grayscale = grayscale, enforce_detection = enforce_detection)
448  	
449  	#--------------------------
450  	
451  	if img.shape[0] > 0 and img.shape[1] > 0:
452  		img = align_face(img = img, detector_backend = detector_backend)
453  	else:
454  		
455  		if enforce_detection == True:
456  			raise ValueError("Detected face shape is ", img.shape,". Consider to set enforce_detection argument to False.")
457  		else: #restore base image 
458  			img = base_img.copy()
459  		
460  	#--------------------------
461  	
462  	#post-processing
463  	if grayscale == True:
464  		img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
465  		
466  	img = cv2.resize(img, target_size)
467  	img_pixels = image.img_to_array(img)
468  	img_pixels = np.expand_dims(img_pixels, axis = 0)
469  	img_pixels /= 255 #normalize input in [0, 1]
470  	
471  	return img_pixels

473  def find_input_shape(model):
474  	
475  	#face recognition models have different size of inputs
476  	#my environment returns (None, 224, 224, 3) but some people mentioned that they got [(None, 224, 224, 3)]. I think this is because of version issue.
477  	
478  	input_shape = model.layers[0].input_shape
479  	
480  	if type(input_shape) == list:
481  		input_shape = input_shape[0][1:3]
482  	else:
483  		input_shape = input_shape[1:3]
484  	
485  	return input_shape
